<file>         ::= <states> "\n" <alphabet> "\n" <transitions>
<states>       ::= "states:" <state> {"," <state>}
<alphabet>     ::= "alphabet:" <symbol> {"," <symbol>}
<transitions>  ::= (<start_state> | <final_states> | <transition>) "\n" {(<start_state> | <final_states> | <transition>) "\n"}
<start_state>  ::= "start:" <state>
<final_states> ::= "final:" <state> {"," <state>}
<transition>   ::= <state> "," <symbol> "->" <state>

<state>        ::= <identifier>
<symbol>       ::= <identifier>
<identifier>   ::= {letter | digit | "_"}+
